#!/bin/bash # Fungsi untuk warna RED='\033[0;31m' GREEN='\033[0;32m' YELLOW='\033[1;33m' BLUE='\033[0;34m' NC='\033[0m' # No Color # Fungsi untuk menampilkan pesan error() { echo -e "${RED}[ERROR]${NC} $1" > /dev/tty } success() { echo -e "${GREEN}[SUCCESS]${NC} $1" > /dev/tty } info() { echo -e "${BLUE}[INFO]${NC} $1" > /dev/tty } warning() { echo -e "${YELLOW}[WARNING]${NC} $1" > /dev/tty } # Fungsi validasi validate_verus_address() { local address=$1 [[ $address =~ ^R[a-zA-Z0-9]{33}$ ]] return $? } validate_url() { local url=$1 [[ $url =~ ^[a-zA-Z0-9.-]+\.[a-zA-Z]{2,}$ ]] return $? } validate_port() { local port=$1 [[ "$port" =~ ^[0-9]+$ ]] && [ "$port" -ge 1 ] && [ "$port" -le 65535 ] return $? } # Fungsi cleanup cleanup() { local exit_code=$? if [ $exit_code -ne 0 ]; then error "Script terminated with error code $exit_code" fi exit $exit_code } # Set trap trap cleanup EXIT trap 'error "Script interrupted by user"; exit 1' INT clear info "Starting setup..." # Update packages info "Updating packages..." if ! yes | pkg update && pkg upgrade && apt update && apt upgrade; then error "Failed to update packages" exit 1 fi success "Packages updated successfully" # Install required packages info "Installing required packages..." if ! yes | pkg install libjansson wget nano git; then error "Failed to install required packages" exit 1 fi success "Required packages installed successfully" # Change directory cd $HOME || exit 1 clear # Check existing ccminer directory if [ -d "$HOME/ccminer" ]; then warning "CCMiner directory already exists!" echo -n "Do you want to delete and replace it? (Y/n): " > /dev/tty read -r replace_choice < /dev/tty replace_choice=${replace_choice:-Y} if [ "${replace_choice^^}" = "Y" ]; then rm -rf "$HOME/ccminer" success "Old CCMiner directory removed" else error "Setup cancelled by user" exit 1 fi fi # Clone repository info "Cloning CCMiner repository..." if ! git clone https://github.com/alpian9890/ccminer.git; then error "Failed to clone repository" exit 1 fi cd "$HOME/ccminer" || exit 1 # Set permissions chmod +x "$HOME/ccminer" "$HOME/ccminer/"* success "CCMiner cloned and permissions set" # Autorun configuration echo -n "Do you want to enable autorun mining when Termux starts? (Y/n): " > /dev/tty read -r autorun_choice < /dev/tty autorun_choice=${autorun_choice:-Y} if [ "${autorun_choice^^}" = "Y" ]; then echo "cd $HOME/ccminer/&&./start.sh" >> /data/data/com.termux/files/usr/etc/bash.bashrc autorun_status="yes" else autorun_status="no" fi # Pool configuration echo -e "\nChoose pool configuration method:" > /dev/tty echo "1) Full URL (example: stratum+tcp://ap.luckpool.net:3956)" > /dev/tty echo "2) Separate server and port" > /dev/tty echo -n "Select option (1/2): " > /dev/tty read -r url_choice < /dev/tty if [ "$url_choice" = "1" ]; then while true; do echo -n "Enter full pool URL: " > /dev/tty read -r full_url < /dev/tty if [[ $full_url =~ ^stratum\+tcp://[a-zA-Z0-9.-]+\.[a-zA-Z]{2,}:[0-9]+$ ]]; then server_url=$(echo "$full_url" | sed 's|stratum+tcp://\([^:]*\):.*|\1|') port=$(echo "$full_url" | sed 's|.*:\([0-9]*\)$|\1|') break else error "Invalid URL format. Please use format: stratum+tcp://server:port" fi done else while true; do echo -n "Enter server/domain (default: ap.luckpool.net): " > /dev/tty read -r server_url < /dev/tty server_url=${server_url:-ap.luckpool.net} if validate_url "$server_url"; then break else error "Invalid server format" fi done while true; do echo -n "Enter port (default: 3956): " > /dev/tty read -r port < /dev/tty port=${port:-3956} if validate_port "$port"; then break else error "Invalid port number (must be between 1-65535)" fi done fi # Wallet configuration while true; do echo -n "Enter your Veruscoin wallet address: " > /dev/tty read -r wallet_address < /dev/tty if validate_verus_address "$wallet_address"; then break else error "Invalid Verus address format (must start with R and be 34 characters long)" fi done # Worker configuration while true; do echo -n "Enter worker name (default: worker1): " > /dev/tty read -r worker_name < /dev/tty worker_name=${worker_name:-worker1} if [[ $worker_name =~ ^[a-zA-Z0-9_-]+$ ]]; then break else error "Invalid worker name (use only letters, numbers, underscore, and hyphen)" fi done # CPU configuration total_cores=$(nproc) while true; do echo -n "Enter number of CPU threads to use (max: $total_cores): " > /dev/tty read -r cpu_threads < /dev/tty cpu_threads=${cpu_threads:-$total_cores} if [[ "$cpu_threads" =~ ^[0-9]+$ ]] && [ "$cpu_threads" -le "$total_cores" ] && [ "$cpu_threads" -gt 0 ]; then break else error "Invalid number of threads (must be between 1 and $total_cores)" fi done # Create config file cat > config.json << EOL { "pools": [ { "name": "${server_url}", "url": "stratum+tcp://${server_url}:${port}", "timeout": 180, "disabled": 1 } ], "user": "${wallet_address}.${worker_name}", "pass": "", "algo": "verus", "threads": ${cpu_threads}, "cpu-priority": 1, "cpu-affinity": -1, "retry-pause": 10, "api-allow": "192.168.0.0/16", "api-bind": "0.0.0.0:4068" } EOL # Final messages success "Setup completed successfully!" info "Autorun mining: $autorun_status" if [ "$autorun_status" = "no" ]; then info "Since you chose not to autorun, to start mining manually run:" echo -e "${GREEN}cd $HOME/ccminer && ./start.sh${NC}" > /dev/tty fi
